## CSharpExercise

<CSharpExercise
  title="Summe zweier Zahlen"
  task="Lies zwei Zahlen von stdin (je eine Zeile) und gib ihre Summe aus."
  starter={`using System;

class Program

    }

  solution={`using System;

class Program
{
    static void Main()
    {
        int a = int.Parse(Console.ReadLine());

        Console.WriteLine(a + b);
    }
}`}
